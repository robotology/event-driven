cmake_minimum_required(VERSION 2.6)

SET(PROJECTNAME viewer)
PROJECT(${PROJECTNAME})

FIND_PACKAGE(YARP)
#FIND_PACKAGE(ICUB)

set(CMAKE_MODULE_PATH ${YARP_MODULE_PATH} ${ICUB_MODULE_PATH} ${CMAKE_MODULE_PATH})
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${ICUB_DIR}/conf ${YARP_DIR}/conf)

INCLUDE_DIRECTORIES(${PROJECT_SOURCE_DIR} ${YARP_INCLUDE_DIRS})

FILE(GLOB folder_source ${PROJECT_SOURCE_DIR}/*.cpp)
FILE(GLOB folder_header ${PROJECT_SOURCE_DIR}/*.h)

SOURCE_GROUP("Source Files" FILES ${folder_source})
SOURCE_GROUP("Header Files" FILES ${folder_header})

ADD_DEFINITIONS(-D_CRT_SECURE_NO_WARNINGS)

ADD_EXECUTABLE(${PROJECTNAME} ${folder_source} ${folder_header})

TARGET_LINK_LIBRARIES(${PROJECTNAME} ${YARP_LIBRARIES})

INSTALL(TARGETS ${PROJECTNAME} DESTINATION ${PROJECT_SOURCE_DIR}/bin)
