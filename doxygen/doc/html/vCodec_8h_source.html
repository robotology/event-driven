318 template <typename T> inline int countTime(const T &q, int &p_time)
319 {
320  if(!p_time) p_time = q.front().stamp;
321  int dt = q.back().stamp - q.front().stamp;
322  p_time = q.back().stamp;
323  if(dt < 0) dt += vtsHelper::max_stamp;
324  return dt;
325 }
