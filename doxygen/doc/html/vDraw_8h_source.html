134  inline void pttr(int &x, int &y, int &z) {
135  // we want a negative rotation around the y axis (yaw)
136  // a positive rotation around the x axis (pitch) (no roll)
137  // the z should always be negative values.
138  // the points need to be shifted across by negligble amount
139  // the points need to be shifted up by (x = max, y = 0, ts = 0 rotation)
140 
141  int xmod = x*CY + z*SY + 0.5; // +0.5 rounds rather than floor
142  int ymod = y*CX - SX*(-x*SY + z*CY) + 0.5;
143  int zmod = y*SX + CX*(-x*SY + z*CY) + 0.5;
144  x = xmod; y = ymod; z = zmod;
145  }
